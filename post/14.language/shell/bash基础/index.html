${var:-value}         # 若var已设置，则不进行赋值， 否则，var=value

${var##pattern}       # 从var左边去掉最长的pattern
${word:pos:size}      # chars at pos(>0,from left; <0, from right ) len of size
${var/ABC/DEFG}       # replace ABC with DEFG

str="abcdef"
expr length $str   # 4
## 获取字符串长度
echo ${#str}       # 4
## 获取子串index
expr index $str "b"  # 2
## 子串
echo ${str:2:3}    #从第二个位置开始提取3个字符, bcd

## 一个井号(#) 表示从左边截取掉最短的匹配 
## 两个井号(##) 表示从左边截取掉最长的匹配
str="abbc,def,ghi,abcjkl"
echo ${str#a*c}     # 输出,def,ghi,abcjkl  (这里把abbc字串去掉）
echo ${str##a*c}    # 输出jkl，(这里把abbc,def,ghi,abc字串去掉)
echo ${str#"a*c"}   # 输出abbc,def,ghi,abcjkl 因为str中没有"a*c"子串
echo ${str##"a*c"}  # 输出abbc,def,ghi,abcjkl 同理
echo ${str#*a*c*}   # 空
echo ${str##*a*c*}  # 空
echo ${str#d*f}     # 输出abbc,def,ghi,abcjkl, 
echo ${str#*d*f}    # 输出,ghi,abcjkl   

## 一个百分号(%)表示从右边截取最短的匹配
## 两个百分号表示(%%)表示从右边截取最长的匹配
echo ${str%a*l}     # abbc,def,ghi   
echo ${str%%b*l}    # a             
echo ${str%a*c}     # abbc,def,ghi,abcjkl


## 子串替换
str="apple, tree, apple tree"

## /替换第一次出现的apple
echo ${str/apple/APPLE}   # APPLE, tree, apple tree
## //替换所有apple
echo ${str//apple/APPLE}  # APPLE, tree, APPLE tree 
## /#开头匹配替换
echo ${str/#apple/APPLE}  # APPLE, tree, apple tree
## %末尾匹配替换
echo ${str/%apple/APPLE}  # 如果字符串str以apple结尾，则用APPLE替换它

## bash version > 4.1.2

## 字典声明
declare -A hashmap

hashmap=( ['a']=1
       ['b']=2
       ['c']=3 )

## 字典赋值
hashmap["key"]="value"
hashmap["key2"]="value2"
## 获取value
echo ${hashmap["key"]}
## 所有key
echo ${!hashmap[@]}
## 所有value
echo ${hashmap[@]} 

## 
hashmap+=( ["key4"]=val4 ["key5"]=val5 )
## 删除key
unset host_panes[$delhost]

## 遍历key
for key in ${!hashmap[@]}; do echo $key ${hashmap[$key]}; done
## 遍历value
for value in ${hashmap[@]}; do echo $value; done

## 关联数组作为函数参数传递
function print_array {
    # eval string into a new associative array
    eval "declare -A func_assoc_array="${1#*=}
    # proof that array was successfully created
    declare -p func_assoc_array
}

# declare an associative array
declare -A assoc_array=(["key1"]="value1" ["key2"]="value2")
# show associative array definition
declare -p assoc_array

# pass associative array in string form to function
print_array "$(declare -p assoc_array)"
